 \es{Principio de máxima incertidumbre: contar caminos}
 \end{textblock}
\vspace{1.25cm} 


\only<1>{
\begin{center}
 \en{\Large Count all the ways data can happen \\ according to asummptions (Hypotesis and Model)}
 \es{\Large Contar todos los caminos que pueden generar los datos \\ de acuerdo a los supuestos: hipótesis y modelo}
\end{center}

\vspace{0.3cm}

\begin{center}
 \en{\Large Hypotesis with more ways that are consistent \\ with data and the model are more plausible}
 \es{\Large Las hipótesis con más caminos consistente \\ con los datos y el modelo son los más plausibles}
\end{center}
}

\only<2>{

 \begin{equation*}
  \text{\en{Belief}\es{Creencia}}(H|D,M) = \frac{\text{\en{Number of ways $D$ can happen given $H$ and $M$}\es{Cantidad de caminos que pueden generar $D$ dado $H$ y $M$}}}{\text{\en{Total number of paths}\es{Cantidad de caminos totales}}}
 \end{equation*}
